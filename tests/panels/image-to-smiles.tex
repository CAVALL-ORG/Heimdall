\begin{abstract}
\noindent
This panel is the 16-row representative slice of
\texttt{tests/ketcher/image-to-smiles/manifest.jsonl} used as a quick
regression sweep for the \texttt{ketcher-image-rebuild} skill. Each row
probes a distinct failure surface --- baseline aromatic, drawn-H +
charges, wedge CIP, $E$/$Z$ encoding, adversarial filename, refusal,
$\alpha$/$\beta$ anomer, multi-charge, macrocycle topology, and the
tier-2 dense-stereo cluster (PLAN-a004-class-robustness). Together they
test reconstruction capability across the full complexity range from
6-heavy aromatic to 62-heavy fused taxane (paclitaxel) without paying
for the 98-row full image manifest. Triggered by
\texttt{Run image-to-smiles panel per the runbook} or
\texttt{/test image-to-smiles-panel}. Panel definition (id list +
trigger phrase) lives in \texttt{tests/AGENT\_RUNBOOK.md} \S Panels;
this PDF is the typeset companion.
\end{abstract}

\section*{Panel members}

\panelrow{I001}{Benzene}{S9-clean}{clean/benzene_clean.png}{%
  \rationale{%
    Baseline aromatic round-trip --- smallest non-trivial fixture.
  }{%
    Kekul\'e $\rightarrow$ aromatize $\rightarrow$ export pipeline on a
    perfectly symmetric arene. If this fails, nothing downstream works.
  }{%
    heavy=6, rings=1 aromatic, wedges=0, charges=0, drawn\_H=0.
  }
}

\panelrow{I007}{Glycine zwitterion}{S9-clean}{clean/glycine_zwitterion_clean.png}{%
  \rationale{%
    Drawn-H plus formal-charge preservation under canonicalization.
  }{%
    Silent neutralization on export --- RDKit canonicalization erases
    the zwitterion, so any pipeline that round-trips through canonical
    loses the \texttt{N+}/\texttt{O-} pair.
  }{%
    heavy=5, rings=0, charges=2 (\texttt{N+}/\texttt{O-}), drawn\_H on N.
  }
}

\panelrow{I008}{L-phenylalanine}{S9-clean}{clean/l_phenylalanine_clean.png}{%
  \rationale{%
    Single-wedge CIP at C$_\alpha$; ``(S)'' label printed at the centre.
  }{%
    May 2026 \texttt{set\_wedge\_bond} L$\leftrightarrow$D inversion ---
    when the chiral atom is not made \texttt{bond.begin}, exported CIP
    silently flips L to D.
  }{%
    heavy=12, rings=1 aromatic, wedges=1, charges=0.
  }
}

\panelrow{I009}{cis-stilbene}{S9-clean}{clean/cis_stilbene_clean.png}{%
  \rationale{%
    $E$/$Z$ encoding on the central C=C (both phenyls same side,
    ``(Z)'' label visible).
  }{%
    First-guess defaults to trans-stilbene from rote memory; the row
    catches that single-shot inversion.
  }{%
    heavy=14, rings=2 aromatic, wedges=0, \texttt{cis\_trans=1}.
  }
}

\panelrow{I011}{Caffeine}{S9b-noisy}{noisy/rotated_not_aspirin.png}{%
  \rationale{%
    Filename-leakage resistance --- file is called
    \texttt{rotated\_not\_aspirin.png}, drawn molecule is caffeine.
  }{%
    Any pipeline that short-circuits to the filename token rather than
    transcribing the pixels exports aspirin instead.
  }{%
    heavy=14, rings=2 (6+5 fused purine), 2 C=O, 3 N-methyls.
  }
}

\panelrow{I017}{Phenol}{S9c-adversarial}{adversarial/phenol_but_filename_says_toluene.png}{%
  \rationale{%
    Second filename-leakage fixture (different compound pair) ---
    file says toluene, drawn molecule is phenol.
  }{%
    Generalization check: confirms the filename-refusal behaviour does
    not depend on the specific caffeine/aspirin pairing in I011.
  }{%
    heavy=7, rings=1 aromatic, OH at one ring position.
  }
}

\panelrow{N001}{Negative control --- reaction arrow only}{S10-negative}{negative/reaction_arrow_only.png}{%
  \rationale{%
    Refusal --- image is a bare arrow on blank background; no atoms,
    bonds, rings, or labels.
  }{%
    ``Hallucinate a SMILES anyway'' failure mode. Skill must emit
    ``Recognition failed.'' and \emph{must not} call
    \texttt{load\_smiles}, \texttt{load\_canonical}, or
    \texttt{build\_from\_graph}.
  }{%
    No chemistry. Trace must contain
    \texttt{handle\_recognition\_failure\_without\_invention}.
  }
}

\panelrow{RT001}{cis-stilbene (round-trip regression)}{S13-roundtrip}{clean/cis_stilbene_clean.png}{%
  \rationale{%
    Stage R readback-loop catches the same trans$\rightarrow$cis flip
    as I009, but under the \texttt{image\_roundtrip\_evaluator}
    grading mode.
  }{%
    Verifies the readback iteration corrects a wrong first guess
    \emph{before} export, not only the one-shot build path.
  }{%
    Same source image as I009: heavy=14, \texttt{cis\_trans=1}.
  }
}

\panelrow{A004}{Paclitaxel (taxol)}{S11-academic}{academic/taxol_core.png}{%

\panelrow{A005}{$\alpha$-D-glucopyranose}{S11-academic}{academic/glucose_pyranose.png}{%

  }{%

\panelrowwidecell{A006}{ATP tetraanion}{S11-academic}{academic/atp.png}{%

  }{%
    heavy=31, rings=3 (6arom + 5arom + 5aliph), wedges=4, charges=4,

\panelrow{D055}{Porphine}{S26-macrocycle}{diverse/porphine.png}{%

  }{%
    Pure-topology fixture (no stereo, no charges). N-H tautomer
    placement --- which 2 of 4 nitrogens carry H --- is the
    discriminator.
  }{%
    heavy=24, rings=5 (4$\times$5-arom + 16-arom), wedges=0, charges=0,
    drawn\_H=2.
  }
}

\panelrow{A009}{Vinblastine (tier-2)}{S11-academic}{academic/vinblastine.png}{%
  \rationale{%
    Tier-2 PLAN-a004-class-robustness --- dimeric \emph{Catharanthus}

  }{%
    Extends the paclitaxel dense-stereo class; saddle-cluster transcription

  }{%

  }

\panelrow{A010}{Thymine (image-truth tautomer)}{S22-nucleobase}{diverse/thymine.png}{%
  \rationale{%
    Image-truth tautomer probe --- the drawn structure is the
    2,4-dione lactam (two N--H, two C=O); canonical SMILES would
    unify with the chemically-equivalent lactim (C=N + OH) form
    under InChI / RDKit normalization.
  }{%
    Isolated drawn-tautomer / drawn-C=O-position failure surface.
    D055 (porphine NH-tautomer) is the only other panel row probing
    this axis, and there the tautomer signal is confounded with
    macrocycle complexity. A010 isolates the tautomer probe on the
    smallest, cleanest fixture in the manifest --- if the evaluator
    vision-compare step doesn't enforce drawn tautomer here, it
    won't enforce it anywhere.
  }{%
    heavy=9, rings=1 (6-mem aromatic), K=0, drawn-H=2 (two
    explicit N--H), wedges=0, charges=0.
  }
}

\panelrow{A011}{Hemibrevetoxin B (tier-2)}{S11-academic}{academic/hemibrevetoxin_b.png}{%
  \rationale{%
    Tier-2 PLAN-a004-class-robustness --- marine polyether ladder, four

  }{%
    Ladder-fusion topology slot. Failure mode in prior session was V4

    \texttt{layoutPolicy:ketcher\_clean\_locked} ($Z$$\rightarrow$$E$
    flip).
  }{%
    heavy=35, rings=4 (2$\times$6 + 2$\times$7), wedges=10,

  }
